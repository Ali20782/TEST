name: Test Suite

on:
  push:
    branches: [ main, develop ]
  pull_request:
    branches: [ main, develop ]
  workflow_dispatch:

jobs:
  unit-tests:
    name: Unit Tests
    runs-on: ubuntu-latest
    
    steps:
    - name: Checkout code
      uses: actions/checkout@v3
    
    - name: Set up Python
      uses: actions/setup-python@v4
      with:
        python-version: '3.10'
    
    - name: Cache dependencies
      uses: actions/cache@v3
      with:
        path: ~/.cache/pip
        key: ${{ runner.os }}-pip-${{ hashFiles('requirements.txt') }}
        restore-keys: |
          ${{ runner.os }}-pip-
    
    - name: Install dependencies
      run: |
        python -m pip install --upgrade pip
        pip install -r requirements.txt
    
    - name: Run unit tests
      run: |
        pytest scripts/tests/test_data_processing.py -v --tb=short
        pytest scripts/tests/test_embedding_service.py -v --tb=short
    
    - name: Upload unit test results
      if: always()
      uses: actions/upload-artifact@v3
      with:
        name: unit-test-results
        path: test_results.json

  integration-tests:
    name: Integration Tests
    runs-on: ubuntu-latest
    
    services:
      postgres:
        image: pgvector/pgvector:pg16
        env:
          POSTGRES_USER: process_admin
          POSTGRES_PASSWORD: test_password
          POSTGRES_DB: process_mining
        ports:
          - 5432:5432
        options: >-
          --health-cmd pg_isready
          --health-interval 10s
          --health-timeout 5s
          --health-retries 5
    
    steps:
    - name: Checkout code
      uses: actions/checkout@v3
    
    - name: Set up Python
      uses: actions/setup-python@v4
      with:
        python-version: '3.10'
    
    - name: Install dependencies
      run: |
        python -m pip install --upgrade pip
        pip install -r requirements.txt
    
    - name: Set environment variables
      run: |
        echo "DATABASE_URL=postgresql://process_admin:test_password@localhost:5432/process_mining" >> $GITHUB_ENV
    
    - name: Start API server
      run: |
        uvicorn scripts.api_server:app --host 0.0.0.0 --port 8000 &
        sleep 10
    
    - name: Wait for services
      run: |
        for i in {1..30}; do
          if curl -f http://localhost:8000/health; then
            echo "API is ready"
            break
          fi
          echo "Waiting for API..."
          sleep 2
        done
    
    - name: Run database tests
      run: |
        pytest scripts/tests/test_database.py -v --tb=short
    
    - name: Run API tests
      run: |
        pytest scripts/tests/test_api_server.py -v --tb=short
    
    - name: Run integration tests
      run: |
        pytest scripts/tests/test_integration.py -v --tb=short
    
    - name: Upload integration test results
      if: always()
      uses: actions/upload-artifact@v3
      with:
        name: integration-test-results
        path: test_results.json

  coverage:
    name: Code Coverage
    runs-on: ubuntu-latest
    needs: [unit-tests, integration-tests]
    
    services:
      postgres:
        image: pgvector/pgvector:pg16
        env:
          POSTGRES_USER: process_admin
          POSTGRES_PASSWORD: test_password
          POSTGRES_DB: process_mining
        ports:
          - 5432:5432
        options: >-
          --health-cmd pg_isready
          --health-interval 10s
          --health-timeout 5s
          --health-retries 5
    
    steps:
    - name: Checkout code
      uses: actions/checkout@v3
    
    - name: Set up Python
      uses: actions/setup-python@v4
      with:
        python-version: '3.10'
    
    - name: Install dependencies
      run: |
        python -m pip install --upgrade pip
        pip install -r requirements.txt
    
    - name: Set environment variables
      run: |
        echo "DATABASE_URL=postgresql://process_admin:test_password@localhost:5432/process_mining" >> $GITHUB_ENV
    
    - name: Start API server
      run: |
        uvicorn scripts.api_server:app --host 0.0.0.0 --port 8000 &
        sleep 10
    
    - name: Run tests with coverage
      run: |
        pytest scripts/tests/ --cov=scripts --cov-report=xml --cov-report=html
    
    - name: Upload coverage to Codecov
      uses: codecov/codecov-action@v3
      with:
        files: ./coverage.xml
        flags: unittests
        name: codecov-umbrella
        fail_ci_if_error: false
    
    - name: Upload coverage HTML
      uses: actions/upload-artifact@v3
      with:
        name: coverage-report
        path: htmlcov/

  full-test-suite:
    name: Full Test Suite
    runs-on: ubuntu-latest
    needs: [unit-tests, integration-tests]
    
    steps:
    - name: Checkout code
      uses: actions/checkout@v3
    
    - name: Set up Python
      uses: actions/setup-python@v4
      with:
        python-version: '3.10'
    
    - name: Install dependencies
      run: |
        python -m pip install --upgrade pip
        pip install -r requirements.txt
    
    - name: Start Docker Compose
      run: |
        docker-compose up -d
        sleep 30
    
    - name: Check services
      run: |
        docker-compose ps
        curl http://localhost:8000/health
    
    - name: Run full test suite
      run: |
        python scripts/tests/run_tests.py all
    
    - name: Upload test results
      if: always()
      uses: actions/upload-artifact@v3
      with:
        name: full-test-results
        path: test_results.json
    
    - name: Stop Docker Compose
      if: always()
      run: |
        docker-compose logs
        docker-compose down -v
